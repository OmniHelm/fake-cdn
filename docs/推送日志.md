#### 请求Body参数
```javascript
{
  "tenantId":"cug", // 租户id
  "start_time": 1741169158000, // 时间戳
  "country": "cn", // 国家
  "region": "mainland_china", // 地区
  "domain": "example.com", // 域名
  "bw": 1024, // 带宽
  "flux": 2048, // 流量
  "bs_bw": 512, // 回源带宽
  "bs_flux": 1024, // 回源流量
  "req_num": 1000, // 请求总数
  "hit_num": 800, // 命中请求数
  "bs_num": 200, // 回源请求数
  "bs_fail_num": 10, // 回源失败请求数
  "hit_flux": 1600, // 命中流量
  "http_code_2xx": 750, // HTTP 2XX 状态码数量
  "http_code_3xx": 50, // HTTP 3XX 状态码数量
  "http_code_4xx": 150, // HTTP 4XX 状态码数量
  "http_code_5xx": 50, // HTTP 5XX 状态码数量
  "bs_http_code_2xx": 180, // 回源 HTTP 2XX 状态码数量
  "bs_http_code_3xx": 10, // 回源 HTTP 3XX 状态码数量
  "bs_http_code_4xx": 8, // 回源 HTTP 4XX 状态码数量
  "bs_http_code_5xx": 2, // 回源 HTTP 5XX 状态码数量
  "interval": 300 // 时间粒度  300(5分钟) / 3600(1小时) / 86400(1天) / 2592000(1月)
}
```

bs_num | 200 | Integer | 是 | 回源请求数
bs_fail_num | 10 | Integer | 是 | 回源失败请求数
hit_flux | 1600 | Integer | 是 | 命中流量
http_code_2xx | 750 | Integer | 是 | HTTP 2XX 状态码数量
http_code_3xx | 50 | Integer | 是 | HTTP 3XX 状态码数量
http_code_4xx | 150 | Integer | 是 | HTTP 4XX 状态码数量
http_code_5xx | 50 | Integer | 是 | HTTP 5XX 状态码数量
